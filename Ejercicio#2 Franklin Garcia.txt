(define (verificarN N L)
(cond ((null? L) 0)
((eq? N (car L)) (+ 1 (verificarN N (cdr L))))
(else (verificarN N (cdr L)))))
(define deleteN
(lambda (item list)
(cond
((null? list) '())
((equal? item (car list)) (deleteN item (cdr list)))
(else (cons (car list) (deleteN item (cdr list)))))))
